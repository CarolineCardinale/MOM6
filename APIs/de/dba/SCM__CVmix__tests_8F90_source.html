48 !> Initializes temperature and salinity for the SCM CVmix test example
49 subroutine scm_cvmix_tests_ts_init(T, S, h, G, GV, param_file)
50  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(out) :: T !< Potential tempera\ture (degC)
51  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(out) :: S !< Salinity (psu)
52  real, dimension(NIMEM_,NJMEM_, NKMEM_), intent(in) :: h !< Layer thickness (m or Pa)
53  type(ocean_grid_type), intent(in) :: G !< Grid structure
54  type(verticalgrid_type), intent(in) :: GV!< Vertical grid structure
55  type(param_file_type), intent(in) :: param_file !< Input parameter structure
56  ! Local variables
57  real :: eta(szk_(g)+1) ! The 1-d nominal positions of the interfaces.
58  real :: UpperLayerTempMLD !< Upper layer Temp MLD thickness (m)
59  real :: UpperLayerSaltMLD !< Upper layer Salt MLD thickness (m)
60  real :: UpperLayerTemp !< Upper layer temperature (SST if thickness 0) (deg C)
61  real :: UpperLayerSalt !< Upper layer salinity (SSS if thickness 0) (PPT)
62  real :: LowerLayerTemp !< Temp at top of lower layer (deg C)
63  real :: LowerLayerSalt !< Salt at top of lower layer (PPT)
64  real :: LowerLayerdTdz !< Temp gradient in lower layer (deg C m^{-1})
65  real :: LowerLayerdSdz !< Salt gradient in lower layer (PPT m^{-1})
66  real :: zC, DZ
67  integer :: i, j, k, is, ie, js, je, isd, ied, jsd, jed, nz
68 
69  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
70  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
71 
72 
73  call log_version(param_file, mod, version)
74  call get_param(param_file,mod,"SCM_TEMP_MLD",upperlayertempmld, &
75  'Initial temp mixed layer depth', units='m',default=0.0)
76  call get_param(param_file,mod,"SCM_SALT_MLD",upperlayersaltmld, &
77  'Initial salt mixed layer depth', units='m',default=0.0)
78  call get_param(param_file,mod,"SCM_L1_SALT",upperlayersalt, &
79  'Layer 2 surface salinity', units='1e-3',default=35.0)
80  call get_param(param_file,mod,"SCM_L1_TEMP",upperlayertemp, &
81  'Layer 1 surface temperature', units='C', default=20.0)
82  call get_param(param_file,mod,"SCM_L2_SALT",lowerlayersalt, &
83  'Layer 2 surface salinity', units='1e-3',default=35.0)
84  call get_param(param_file,mod,"SCM_L2_TEMP",lowerlayertemp, &
85  'Layer 2 surface temperature', units='C', default=20.0)
86  call get_param(param_file,mod,"SCM_L2_DTDZ",lowerlayerdtdz, &
87  'Initial temperature stratification in layer 2', &
88  units='C/m', default=0.00)
89  call get_param(param_file,mod,"SCM_L2_DSDZ",lowerlayerdsdz, &
90  'Initial salinity stratification in layer 2', &
91  units='PPT/m', default=0.00)
92 
93  do j=js,je ; do i=is,ie
94  eta(1) = 0. ! Reference to surface
95  do k=1,nz
96  eta(k+1) = eta(k) - h(i,j,k)*gv%H_to_m ! Interface below layer (in m)
97  zc = 0.5*( eta(k) + eta(k+1) ) ! Z of middle of layer (in m)
98  dz = min(0., zc+upperlayertempmld)
99  if (dz.ge.0.0) then ! in Layer 1
100  t(i,j,k) = upperlayertemp
101  else ! in Layer 2
102  t(i,j,k) = lowerlayertemp + lowerlayerdtdz * dz
103  endif
104  dz = min(0., zc+upperlayersaltmld)
105  if (dz.ge.0.0) then ! in Layer 1
106  s(i,j,k) = upperlayersalt
107  else ! in Layer 2
108  s(i,j,k) = lowerlayersalt + lowerlayerdsdz * dz
109  endif
110  enddo ! k
111  enddo ; enddo
112 
113 end subroutine scm_cvmix_tests_ts_init

241  do j=jsq,jeq ; do i=is,ie

247  do j=jsq,jeq ; do i=is,ie
248  ! Note CVmix test inputs give evaporation in m/s

251  fluxes%evap(i,j) = cs%surf_evap * cs%Rho0
252  enddo; enddo
253  endif
254 
255  if (cs%UseDiurnalSW) then
256  do j=jsq,jeq ; do i=is,ie
257  ! Note CVmix test inputs give max sw rad in [m K/s]
258  ! therefore must convert to W/m2 by multiplying
259  ! by Rho0*Cp
260  ! Note diurnal cycle peaks at Noon.
261  fluxes%sw(i,j) = cs%Max_sw * max(0.0,cos(2*pi* &
262  (time_type_to_real(day)/86400.-0.5))) * cs%RHO0 * fluxes%C_p
