371 subroutine call_tracer_column_fns(h_old, h_new, ea, eb, fluxes, dt, G, GV, tv, optics, CS, &
372  debug, evap_CFL_limit, minimum_forcing_depth)
373  real, dimension(NIMEM_,NJMEM_,NKMEM_), intent(in) :: h_old, h_new, ea, eb
374  type(forcing), intent(in) :: fluxes
375  real, intent(in) :: dt
376  type(ocean_grid_type), intent(in) :: G
377  type(verticalgrid_type), intent(in) :: GV
378  type(thermo_var_ptrs), intent(in) :: tv
379  type(optics_type), pointer :: optics
380  type(tracer_flow_control_cs), pointer :: CS

382  real, optional,intent(in) :: evap_CFL_limit

384 
385 ! This subroutine calls all registered tracer column physics
386 ! subroutines.
387 
388 ! Arguments: h_old - Layer thickness before entrainment, in m (Boussinesq)
389 ! or kg m-2 (non-Boussinesq).
390 ! (in) h_new - Layer thickness after entrainment, in m or kg m-2.
391 ! (in) ea - an array to which the amount of fluid entrained
392 ! from the layer above during this call will be
393 ! added, in m or kg m-2, the same as h_old.
394 ! (in) eb - an array to which the amount of fluid entrained
395 ! from the layer below during this call will be
396 ! added, in m or kg m-2, the same as h_old.
397 ! (in) fluxes - A structure containing pointers to any possible
398 ! forcing fields. Unused fields have NULL ptrs.
399 ! (in) dt - The amount of time covered by this call, in s.
400 ! (in) G - The ocean's grid structure.
401 ! (in) GV - The ocean's vertical grid structure.
402 ! (in) tv - The structure containing thermodynamic variables.
403 ! (in) optics - The structure containing optical properties.
404 ! (in) CS - The control structure returned by a previous call to
405 ! call_tracer_register.
