26 !* This file contains the subroutine (energetic_PBL) that uses an *
27 !* integrated boundary layer energy budget (like a bulk- or refined- *
28 !* bulk mixed layer scheme), but instead of homogenizing this model *
29 !* calculates a finite diffusivity and viscosity, which in this *
30 !* regard is conceptually similar to what is done with KPP or various *
31 !* two-equation closures. However, the scheme that is implemented *
32 !* here has the big advantage that is entirely implicit, but is *
33 !* simple enough that it requires only a single vertical pass to *
34 !* determine the diffusivity. The development of bulk mixed layer *
35 !* models stems from the work of various people, as described in the *
36 !* review paper by Niiler and Kraus (1979). The work here draws in *
37 !* with particular on the form for TKE decay proposed by Oberhuber *
38 !* (JPO, 1993, 808-829), with an extension to a refined bulk mixed *
39 !* layer as described in Hallberg (Aha Huliko'a, 2003). The physical *
40 !* processes portrayed in this subroutine include convectively driven *
41 !* mixing and mechanically driven mixing. Unlike boundary-layer *
42 !* mixing, stratified shear mixing is not a one-directional turbulent *
43 !* process, and it is dealt with elsewhere in the MOM6 code within *
44 !* the module MOM_kappa_shear.F90. It is assumed that the heat, *
45 !* mass, and salt fluxes have been applied elsewhere, but that their *
46 !* implications for the integrated TKE budget have been captured in *
47 !* an array that is provided as an argument to this subroutine. This *
48 !* is a full 3-d array due to the effects of penetrating shortwave *
49 !* radiation. *
50 !* *
51 !* Macros written all in capital letters are defined in MOM_memory.h. *
52 !* *
53 !* A small fragment of the grid is shown below: *
54 !* *
55 !* j+1 x ^ x ^ x At x: q *
56 !* j+1 > o > o > At ^: v *
57 !* j x ^ x ^ x At >: u *
58 !* j > o > o > At o: h, T, S, Kd, etc. *
59 !* j-1 x ^ x ^ x *
60 !* i-1 i i+1 At x & ^: *
61 !* i i+1 At > & o: *
62 !* *
63 !* The boundaries always run through q grid points (x). *
64 !* *

198 ! This subroutine determines the diffusivities from the integrated energetics
199 ! mixed layer model. It assumes that heating, cooling and freshwater fluxes
200 ! have already been applied. All calculations are done implicitly, and there
201 ! is no stability limit on the time step.
202 !
203 ! For each interior interface, first discard the TKE to account for mixing
204 ! of shortwave radiation through the next denser cell. Next drive mixing based
205 ! on the local? values of ustar + wstar, subject to available energy. This
206 ! step sets the value of Kd(K). Any remaining energy is then subject to decay
207 ! before being handed off to the next interface. mech_TKE and conv_PErel are treated
208 ! separately for the purposes of decay, but are used proportionately to drive
209 ! mixing.
210 !
211 ! The key parameters for the mixed layer are found in the control structure.
212 ! These include mstar, nstar, TKE_decay, and conv_decay. For the Oberhuber (1993) mixed layer,
213 ! the values of these are:
214 ! mstar = 1.25, nstar = 1, TKE_decay = 2.5, conv_decay = 0.5
215 ! TKE_decay is 1/kappa in eq. 28 of Oberhuber (1993), while
216 ! conv_decay is 1/mu.
217 ! For a traditional Kraus-Turner mixed layer, the values are:
218 ! mstar = 1.25, nstar = 0.4, TKE_decay = 0.0, conv_decay = 0.0

831  dse(k-1) = b1 * ( dse_t2 )

838  ds_to_dcolht_a(i,k) = ds_to_dcolht(i,k)

844  ! Kddt_h(K).

867  ! Using Pr=1 and the diffusivity at the bottom interface (once it is
868  ! known), determine how much resolved mean kinetic energy (MKE) will be

879  ! extracted by mixing with a finite viscosity.

881  ((htot(i)+h_neglect) * (h(i,k)+h_neglect))

902  kd_guess0 = vstar * vonkar * mixing_length_used(k)
903  endif
904  else

906  endif
907  vstar_used(k) = vstar ! Track vstar

909 
910  if (cs%orig_PE_calc) then
911  call find_pe_chg_orig(kddt_h_g0, h(i,k), hp_a(i), dte_term, dse_term, &
912  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &

924  pe_chg=pe_chg_g0, dpec_dkd=dpea_dkd_g0, dpe_max=pe_chg_max, &
925  dpec_dkd_0=dpec_dkd_kd0 )

955  endif
956  else
957  vstar = 0.0 ; kd(i,k) = 0.0
958  endif
959  vstar_used(k) = vstar
960 
961  if (cs%orig_PE_calc) then
962  call find_pe_chg_orig(kd(i,k)*dt_h, h(i,k), hp_a(i), dte_term, dse_term, &
963  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &

974  pe_chg=dpe_conv)
975  endif

985  endif

991  endif

993  cs%ML_depth(i,j) = cs%ML_depth(i,j) + gv%H_to_m * h(i,k)
994  !CS%ML_depth2(i,j) = CS%ML_depth2(i,J) + GV%H_to_m * h(i,k)
995  endif
996 
997  kddt_h(k) = kd(i,k)*dt_h
998  elseif (tot_tke + (mke_src - n2_dissipation*pe_chg_g0) >= 0.0) then
999  ! There is energy to support the suggested mixing. Keep that estimate.
1000  kd(i,k) = kd_guess0
1001  kddt_h(k) = kddt_h_g0
1002 
1003  ! Reduce the mechanical and convective TKE proportionately.
1004  tot_tke = tot_tke + mke_src
1005  tke_reduc = 0.0 ! tot_TKE could be 0 if Convectively_stable is false.
1006  if (tot_tke > 0.0) tke_reduc = (tot_tke - n2_dissipation*pe_chg_g0) &
1007  / tot_tke
1008  if (cs%TKE_diagnostics) then
1009  dtke_mixing = dtke_mixing - pe_chg_g0 * idtdr0
1010  dtke_mke = dtke_mke + mke_src * idtdr0
1011  dtke_conv_decay = dtke_conv_decay + &
1012  (1.0-tke_reduc)*(cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
1013  endif
1014  tot_tke = tke_reduc*tot_tke
1015  mech_tke(i) = tke_reduc*(mech_tke(i) + mke_src)
1016  conv_perel(i) = tke_reduc*conv_perel(i)
1017  if (sfc_connected(i)) then
1018  cs%ML_depth(i,j) = cs%ML_depth(i,j) + gv%H_to_m * h(i,k)
1019  !CS%ML_depth2(i,J) = CS%ML_depth2(i,J) + GV%H_to_m * h(i,k)
1020  endif
1021  elseif (tot_tke == 0.0) then
1022  ! This can arise if nstar_FC = 0.
1023  kd(i,k) = 0.0 ; kddt_h(k) = 0.0
1024  tot_tke = 0.0 ; conv_perel(i) = 0.0 ; mech_tke(i) = 0.0
1025  sfc_disconnect = .true.
1026  else
1027  ! There is not enough energy to support the mixing, so reduce the
1028  ! diffusivity to what can be supported.
1029  kddt_h_max = kddt_h_g0 ; kddt_h_min = 0.0
1030  tke_left_max = tot_tke + (mke_src - n2_dissipation*pe_chg_g0) ;
1031  tke_left_min = tot_tke
1032 
1033  ! As a starting guess, take the minimum of a false position estimate
1034  ! and a Newton's method estimate starting from Kddt_h = 0.0.
1035  kddt_h_guess = tot_tke * kddt_h_max / max( n2_dissipation*pe_chg_g0 &
1036  - mke_src, kddt_h_max * (dpec_dkd_kd0 - dmke_max * &
1037  mke2_hharm) )
1038  ! The above expression is mathematically the same as the following
1039  ! except it is not susceptible to division by zero when
1040  ! dPEc_dKd_Kd0 = dMKE_max = 0 .
1041  ! Kddt_h_guess = tot_TKE * min( Kddt_h_max / (PE_chg_g0 - MKE_src), &
1042  ! 1.0 / (dPEc_dKd_Kd0 - dMKE_max * MKE2_Hharm) )
1043  if (debug) then
1044  tke_left_itt(:) = 0.0 ; dpea_dkd_itt(:) = 0.0 ; pe_chg_itt(:) = 0.0
1045  mke_src_itt(:) = 0.0 ; kddt_h_itt(:) = 0.0
1046  endif

1048  if (cs%orig_PE_calc) then
1049  call find_pe_chg_orig(kddt_h_guess, h(i,k), hp_a(i), dte_term, dse_term, &
1050  dt_km1_t2, ds_km1_t2, dt_to_dpe(i,k), ds_to_dpe(i,k), &
1051  dt_to_dpe_a(i,k-1), ds_to_dpe_a(i,k-1), &
1052  pres(i,k), dt_to_dcolht(i,k), ds_to_dcolht(i,k), &
1053  dt_to_dcolht_a(i,k-1), ds_to_dcolht_a(i,k-1), &
1054  pe_chg=pe_chg, dpec_dkd=dpec_dkd )
1055  else

1057  th_a(k-1), sh_a(k-1), th_b(k), sh_b(k), &
1058  dt_to_dpe_a(i,k-1), ds_to_dpe_a(i,k-1), dt_to_dpe(i,k), ds_to_dpe(i,k), &
1059  pres(i,k), dt_to_dcolht_a(i,k-1), ds_to_dcolht_a(i,k-1), &
1060  dt_to_dcolht(i,k), ds_to_dcolht(i,k), &
1061  pe_chg=dpe_conv)
1062  endif
1063  mke_src = dmke_max * (1.0 - exp(-mke2_hharm * kddt_h_guess))
1064  dmke_src_dk = dmke_max * mke2_hharm * exp(-mke2_hharm * kddt_h_guess)
1065 
1066  tke_left = tot_tke + (mke_src - n2_dissipation*pe_chg)
1067  if (debug) then
1068  kddt_h_itt(itt) = kddt_h_guess ; mke_src_itt(itt) = mke_src
1069  pe_chg_itt(itt) = n2_dissipation*pe_chg
1070  tke_left_itt(itt) = tke_left
1071  dpea_dkd_itt(itt) = dpec_dkd
1072  endif
1073  ! Store the new bounding values, bearing in mind that min and max
1074  ! here refer to Kddt_h and dTKE_left/dKddt_h < 0:
1075  if (tke_left >= 0.0) then
1076  kddt_h_min = kddt_h_guess ; tke_left_min = tke_left
1077  else
1078  kddt_h_max = kddt_h_guess ; tke_left_max = tke_left
1079  endif
1080 
1081  ! Try to use Newton's method, but if it would go outside the bracketed
1082  ! values use the false-position method instead.
1083  use_newt = .true.
1084  if (dpec_dkd*n2_dissipation - dmke_src_dk <= 0.0) then
1085  use_newt = .false.
1086  else

1091  endif

1098  (tke_left_max - tke_left_min)

1104  if (debug) num_itts(k) = itt

1108  endif
1109  enddo

1111 
1112  ! All TKE should have been consumed.
1113  if (cs%TKE_diagnostics) then
1114  dtke_mixing = dtke_mixing - (tot_tke + mke_src) * idtdr0
1115  dtke_mke = dtke_mke + mke_src * idtdr0
1116  dtke_conv_decay = dtke_conv_decay + &
1117  (cs%nstar-nstar_fc) * conv_perel(i) * idtdr0
1118  endif
1119 
1120  if (sfc_connected(i)) cs%ML_depth(i,j) = cs%ML_depth(i,j) + &
1121  (pe_chg / pe_chg_g0) * gv%H_to_m * h(i,k)
1122  tot_tke = 0.0 ; mech_tke(i) = 0.0 ; conv_perel(i) = 0.0
1123  sfc_disconnect = .true.
1124  endif
1125 
1126  kddt_h(k) = kd(i,k)*dt_h
1127  ! At this point, the final value of Kddt_h(K) is known, so the
1128  ! estimated properties for layer k-1 can be calculated.
1129  b1 = 1.0 / (hp_a(i) + kddt_h(k))
1130  c1(k) = kddt_h(k) * b1
1131  if (cs%orig_PE_calc) then
1132  dte(k-1) = b1 * ( kddt_h(k)*(t0(k)-t0(k-1)) + dte_t2 )
1133  dse(k-1) = b1 * ( kddt_h(k)*(s0(k)-s0(k-1)) + dse_t2 )
1134  endif
1135

1140  ds_to_dcolht_a(i,k) = ds_to_dcolht(i,k) + c1(k)*ds_to_dcolht_a(i,k-1)
1141 
1142  endif ! tot_TKT > 0.0 branch. Kddt_h(K) has been set.
1143 
1144  ! Store integrated velocities and thicknesses for MKE conversion calculations.
1145  if (sfc_disconnect) then
1146  ! There is no turbulence at this interface, so zero out the running sums.
1147  uhtot(i) = u(i,k)*h(i,k)
1148  vhtot(i) = v(i,k)*h(i,k)
1149  htot(i) = h(i,k)
1150  sfc_connected(i) = .false.
1151  else
1152  uhtot(i) = uhtot(i) + u(i,k)*h(i,k)
1153  vhtot(i) = vhtot(i) + v(i,k)*h(i,k)
1154  htot(i) = htot(i) + h(i,k)
1155  endif
1156 
1157  if (debug) then
1158  if (k==2) then
1159  te(1) = b1*(h(i,1)*t0(1))
1160  se(1) = b1*(h(i,1)*s0(1))
1161  else
1162  te(k-1) = b1 * (h(i,k-1) * t0(k-1) + kddt_h(k-1) * te(k-2))

1164  endif
1165  endif
1166  enddo
1167  kd(i,nz+1) = 0.0
1168

1170  ! Complete the tridiagonal solve for Te.
1171  b1 = 1.0 / hp_a(i)

1178  endif
