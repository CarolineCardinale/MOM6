329 subroutine mom_thermovar_chksum(mesg, tv, G)
330  character(len=*), intent(in) :: mesg
331  type(thermo_var_ptrs), intent(in) :: tv
332  type(ocean_grid_type), intent(in) :: G
333 ! This subroutine writes out chksums for the model's basic state variables.
334 ! Arguments: mesg - A message that appears on the chksum lines.
335 ! (in) u - Zonal velocity, in m s-1.
336 ! (in) v - Meridional velocity, in m s-1.
337 ! (in) h - Layer thickness, in m.
338 ! (in) uh - Volume flux through zonal faces = u*h*dy, m3 s-1.
339 ! (in) vh - Volume flux through meridional faces = v*h*dx, in m3 s-1.
340 ! (in) G - The ocean's grid structure.
341  integer :: is, ie, js, je, nz
342  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
343 
344  ! Note that for the chksum calls to be useful for reproducing across PE
345  ! counts, there must be no redundant points, so all variables use is..ie
346  ! and js...je as their extent.
347  if (associated(tv%T)) &
348  call hchksum(tv%T, mesg//" tv%T",g%HI)
349  if (associated(tv%S)) &
350  call hchksum(tv%S, mesg//" tv%S",g%HI)
351  if (associated(tv%frazil)) &
352  call hchksum(tv%frazil, mesg//" tv%frazil",g%HI)
353  if (associated(tv%salt_deficit)) &
354  call hchksum(tv%salt_deficit, mesg//" tv%salt_deficit",g%HI)
355  if (associated(tv%TempxPmE)) &
356  call hchksum(tv%TempxPmE, mesg//" tv%TempxPmE",g%HI)
357 end subroutine mom_thermovar_chksum
