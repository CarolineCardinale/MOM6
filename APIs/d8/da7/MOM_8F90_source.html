435  type(forcing), intent(inout) :: fluxes !< pointers to forcing fields
436  type(surface), intent(inout) :: state !< surface ocean state
437  type(time_type), intent(in) :: Time_start !< starting time of a segment, as a time type
438  real, intent(in) :: time_interval !< time interval
439  type(mom_control_struct), pointer :: CS !< control structure from initialize_MOM
440 
441  ! local
442  type(ocean_grid_type), pointer :: G ! pointer to a structure containing
443  ! metrics and related information
444  type(verticalgrid_type), pointer :: GV => null()
445  integer, save :: nt = 1 ! running number of iterations
446  integer :: ntstep ! time steps between tracer updates or diabatic forcing
447  integer :: n_max ! number of steps to take in this call
448 
449  integer :: i, j, k, is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz, n
450  integer :: isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
451 
452  real :: dt ! baroclinic time step (sec)
453  real :: dtth ! time step for thickness diffusion (sec)
454  real :: dtdia ! time step for diabatic processes (sec)
455  real :: dt_therm ! a limited and quantized version of CS%dt_therm (sec)
456  real :: dtbt_reset_time ! value of CS%rel_time when DTBT was last calculated (sec)
457 
458  real :: wt_end, wt_beg
459 
460  logical :: calc_dtbt ! Indicates whether the dynamically adjusted
461  ! barotropic time step needs to be updated.
462  logical :: do_advection ! If true, it is time to advect tracers.
463  logical :: thermo_does_span_coupling ! If true, thermodynamic forcing spans
464  ! multiple dynamic timesteps.
465 
466  real, dimension(SZI_(CS%G),SZJ_(CS%G)) :: &
467  eta_av, & ! average sea surface height or column mass over a timestep (meter or kg/m2)
468  ssh ! sea surface height based on eta_av (meter or kg/m2)
469 
470  real, allocatable, dimension(:,:) :: &
471  tmp, & ! temporary 2d field
472  zos, & ! dynamic sea lev (zero area mean) from inverse-barometer adjusted ssh (meter)
473  zossq, & ! square of zos (m^2)
474  sfc_speed, & ! sea surface speed at h-points (m/s)
475  frazil_ave, & ! average frazil heat flux required to keep temp above freezing (W/m2)
476  salt_deficit_ave, & ! average salt flux required to keep salinity above 0.01ppt (gSalt m-2 s-1)
477  Heat_PmE_ave, & ! average effective heat flux into the ocean due to
478  ! the exchange of water with other components, times the
479  ! heat capacity of water, in W m-2.
480  intern_heat_ave ! avg heat flux into ocean from geothermal or
481  ! other internal heat sources (W/m2)
482 
483  real, pointer, dimension(:,:,:) :: &
484  u, & ! u : zonal velocity component (m/s)
485  v, & ! v : meridional velocity component (m/s)
486  h ! h : layer thickness (meter (Bouss) or kg/m2 (non-Bouss))
487 
488  real, dimension(SZI_(CS%G),SZJ_(CS%G),SZK_(CS%G)+1) :: eta_predia, eta_preale
489 
490 
491  real :: tot_wt_ssh, Itot_wt_ssh, I_time_int
492  real :: zos_area_mean, volo, ssh_ga
493  type(time_type) :: Time_local
494  logical :: showCallTree
495  logical :: do_pass_kd_kv_turb ! This is used for a group halo pass.
496  logical :: use_ice_shelf ! Needed for ALE
497  g => cs%G ; gv => cs%GV
498  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
499  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
500  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
501  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
502  u => cs%u ; v => cs%v ; h => cs%h
503 
504  call cpu_clock_begin(id_clock_ocean)
505  call cpu_clock_begin(id_clock_other)
506 
507  showcalltree = calltree_showquery()
508  if (showcalltree) call calltree_enter("step_MOM(), MOM.F90")
509 
510  use_ice_shelf = .false.
511  if (associated(fluxes%frac_shelf_h)) use_ice_shelf = .true.
512 
513  ! First determine the time step that is consistent with this call.
514  ! It is anticipated that the time step will almost always coincide
515  ! with dt. In addition, ntstep is determined, subject to the constraint
516  ! that ntstep cannot exceed n_max.
517  if (time_interval <= cs%dt) then
518  n_max = 1
519  else
520  n_max = ceiling(time_interval/cs%dt - 0.001)
521  endif
522 
523  dt = time_interval / real(n_max)
524  dtdia = 0.0
525  thermo_does_span_coupling = (cs%thermo_spans_coupling .and. &
526  (cs%dt_therm > 1.5*time_interval))
527  if (thermo_does_span_coupling) then
528  ! Set dt_therm to be an integer multiple of the coupling time step.
529  dt_therm = time_interval * floor(cs%dt_therm / time_interval + 0.001)
530  ntstep = floor(dt_therm/dt + 0.001)
531  else
532  ntstep = max(1,min(n_max,floor(cs%dt_therm/dt + 0.001)))
533  dt_therm = dt*ntstep
534  endif
535 
536  cs%visc%calc_bbl = .true.
537 
538  if (.not.ASSOCIATED(fluxes%p_surf)) cs%interp_p_surf = .false.
539 
540  !---------- Begin setup for group halo pass
541 
542  call cpu_clock_begin(id_clock_pass)
543  call create_group_pass(cs%pass_tau_ustar_psurf, fluxes%taux, fluxes%tauy, g%Domain)
544  if (ASSOCIATED(fluxes%ustar)) &
545  call create_group_pass(cs%pass_tau_ustar_psurf, fluxes%ustar(:,:), g%Domain)
546  if (ASSOCIATED(fluxes%p_surf)) &
547  call create_group_pass(cs%pass_tau_ustar_psurf, fluxes%p_surf(:,:), g%Domain)
548  if ((cs%thickness_diffuse .and. (.not.cs%thickness_diffuse_first .or. cs%dt_trans == 0) ) .OR. cs%mixedlayer_restrat) &
549  call create_group_pass(cs%pass_h, h, g%Domain)
550 
551  if (cs%diabatic_first) then
552  if (associated(cs%visc%Ray_u) .and. associated(cs%visc%Ray_v)) &
553  call create_group_pass(cs%pass_ray, cs%visc%Ray_u, cs%visc%Ray_v, g%Domain, &
554  to_all+scalar_pair, cgrid_ne)
555  if (associated(cs%visc%kv_bbl_u) .and. associated(cs%visc%kv_bbl_v)) then
556  call create_group_pass(cs%pass_bbl_thick_kv_bbl, cs%visc%bbl_thick_u, &
557  cs%visc%bbl_thick_v, g%Domain, to_all+scalar_pair, cgrid_ne)
558  call create_group_pass(cs%pass_bbl_thick_kv_bbl, cs%visc%kv_bbl_u, &
559  cs%visc%kv_bbl_v, g%Domain, to_all+scalar_pair, cgrid_ne)
560  endif
561  endif
562 
563  if (.not.cs%adiabatic .AND. cs%use_ALE_algorithm ) then
564  if (cs%use_temperature) then
565  call create_group_pass(cs%pass_T_S_h, cs%tv%T, g%Domain)
566  call create_group_pass(cs%pass_T_S_h, cs%tv%S, g%Domain)
567  endif
568  call create_group_pass(cs%pass_T_S_h, h, g%Domain)
569  endif
570 
571  if ((cs%adiabatic .OR. cs%diabatic_first) .AND. cs%use_temperature) then
572  call create_group_pass(cs%pass_T_S, cs%tv%T, g%Domain)
573  call create_group_pass(cs%pass_T_S, cs%tv%S, g%Domain)
574  endif
575 
576  if ((cs%visc%Prandtl_turb > 0) .and. associated(cs%visc%Kd_turb)) &
577  call create_group_pass(cs%pass_kd_kv_turb, cs%visc%Kd_turb, g%Domain)
578  if (associated(cs%visc%Kv_turb)) &
579  call create_group_pass(cs%pass_kd_kv_turb, cs%visc%Kv_turb, g%Domain)
580 
581  !---------- End setup for group halo pass
582 
583 
584  do_pass_kd_kv_turb = ((cs%visc%Prandtl_turb > 0) .and. associated(cs%visc%Kd_turb)) &
585  .OR. associated(cs%visc%Kv_turb)
586 
587  if (g%nonblocking_updates) then
588  call start_group_pass(cs%pass_tau_ustar_psurf, g%Domain)
589  else
590  call do_group_pass(cs%pass_tau_ustar_psurf, g%Domain)
591  endif
592  call cpu_clock_end(id_clock_pass)
593 
594  if (ASSOCIATED(cs%tv%frazil)) cs%tv%frazil(:,:) = 0.0
595  if (ASSOCIATED(cs%tv%salt_deficit)) cs%tv%salt_deficit(:,:) = 0.0
596  if (ASSOCIATED(cs%tv%TempxPmE)) cs%tv%TempxPmE(:,:) = 0.0
597  if (ASSOCIATED(cs%tv%internal_heat)) cs%tv%internal_heat(:,:) = 0.0
598 
599  cs%rel_time = 0.0
600 
601  tot_wt_ssh = 0.0
602  do j=js,je ; do i=is,ie ; cs%ave_ssh(i,j) = 0.0 ; ssh(i,j) = cs%missing; enddo ; enddo
603 
604  if (associated(cs%VarMix)) then
605  call enable_averaging(time_interval, time_start+set_time(int(time_interval)), &
606  cs%diag)
607  call calc_resoln_function(h, cs%tv, g, gv, cs%VarMix)
608  call disable_averaging(cs%diag)
609  endif
610 
611  if (g%nonblocking_updates) then
612  call cpu_clock_begin(id_clock_pass)
613  call complete_group_pass(cs%pass_tau_ustar_psurf, g%Domain)
614  call cpu_clock_end(id_clock_pass)
615  endif
616 
617  if (cs%interp_p_surf) then
618  if (.not.ASSOCIATED(cs%p_surf_end)) allocate(cs%p_surf_end(isd:ied,jsd:jed))
619  if (.not.ASSOCIATED(cs%p_surf_begin)) allocate(cs%p_surf_begin(isd:ied,jsd:jed))
620  if (.not.cs%p_surf_prev_set) then
621  do j=jsd,jed ; do i=isd,ied
622  cs%p_surf_prev(i,j) = fluxes%p_surf(i,j)
623  enddo ; enddo
624  cs%p_surf_prev_set = .true.
625  endif
626  else
627  cs%p_surf_end => fluxes%p_surf
628  endif
629 
630  if (cs%debug) then
631  call mom_state_chksum("Before steps ", u, v, h, cs%uh, cs%vh, g, gv)
632  call mom_forcing_chksum("Before steps", fluxes, g, haloshift=0)
633  call check_redundant("Before steps ", u, v, g)
634  call check_redundant("Before steps ", fluxes%taux, fluxes%tauy, g)
635  endif
636  call cpu_clock_end(id_clock_other)
637 
638  do n=1,n_max
639 
640  nt = nt + 1
641  call cpu_clock_begin(id_clock_other)
642 
643  ! Set the universally visible time to the middle of the time step
644  cs%Time = time_start + set_time(int(floor(cs%rel_time+0.5*dt+0.5)))
645  cs%rel_time = cs%rel_time + dt
646 
647  ! Set the local time to the end of the time step.
648  time_local = time_start + set_time(int(floor(cs%rel_time+0.5)))
649  if (showcalltree) call calltree_enter("DT cycles (step_MOM) n=",n)
650  call cpu_clock_end(id_clock_other)
651 
652  if (cs%diabatic_first .and. (cs%dt_trans==0.0)) then ! do thermodynamics.
653 
654  if (thermo_does_span_coupling) then
655  dtdia = dt_therm
656  if ((fluxes%dt_buoy_accum > 0.0) .and. (dtdia > time_interval) .and. &
657  (abs(fluxes%dt_buoy_accum - dtdia) > 1e-6*dtdia)) then
658  call mom_error(fatal, "step_MOM: Mismatch between long thermodynamic "//&
659  "timestep and time over which buoyancy fluxes have been accumulated.")
660  endif
661  call mom_error(fatal, "MOM is not yet set up to have restarts that work "//&
662  "with THERMO_SPANS_COUPLING and DIABATIC_FIRST.")
663  else
664  dtdia = dt*min(ntstep,n_max-(n-1))
665  endif
666  ! The end-time of the diagnostic interval needs to be set ahead if there
667  ! are multiple dynamic time steps worth of dynamics applied here.
668  call enable_averaging(dtdia, time_local + &
669  set_time(int(floor(dtdia-dt+0.5))), cs%diag)
670 
671  ! Calculate the BBL properties and store them inside visc (u,h).
672  ! This is here so that CS%visc is updated before diabatic() when
673  ! DIABATIC_FIRST=True. Otherwise diabatic() is called after the dynamics
674  ! and set_viscous_BBL is called as a part of the dynamic stepping.
675 
676  if (cs%debug) then
677  call uchksum(u,"Pre set_viscous_BBL u", g%HI, haloshift=1)
678  call vchksum(v,"Pre set_viscous_BBL v", g%HI, haloshift=1)
679  call hchksum(h*gv%H_to_m,"Pre set_viscous_BBL h", g%HI, haloshift=1)
680  if (associated(cs%tv%T)) call hchksum(cs%tv%T, "Pre set_viscous_BBL T", g%HI, haloshift=1)
681  if (associated(cs%tv%S)) call hchksum(cs%tv%S, "Pre set_viscous_BBL S", g%HI, haloshift=1)
682  endif
683 
684  !call cpu_clock_begin(id_clock_vertvisc)
685  call set_viscous_bbl(u, v, h, cs%tv, cs%visc, g, gv, cs%set_visc_CSp)
686  !call cpu_clock_end(id_clock_vertvisc)
687 
688  call cpu_clock_begin(id_clock_pass)
689  if(associated(cs%visc%Ray_u) .and. associated(cs%visc%Ray_v)) &
690  call do_group_pass(cs%pass_ray, g%Domain )
691  if(associated(cs%visc%kv_bbl_u) .and. associated(cs%visc%kv_bbl_v)) &
692  call do_group_pass(cs%pass_bbl_thick_kv_bbl, g%Domain )
693  call cpu_clock_end(id_clock_pass)
694  if (showcalltree) call calltree_waypoint("done with set_viscous_BBL (step_MOM_dyn_split_RK2)")
695 
696  call cpu_clock_begin(id_clock_thermo)
697 
698  if (.not.cs%adiabatic) then
699 
700  if (cs%debug) then
701  call mom_state_chksum("Pre-dia first ", u, v, h, cs%uhtr, cs%vhtr, g, gv)
702  call mom_thermo_chksum("Pre-dia first ", cs%tv, g, haloshift=0)
703  call check_redundant("Pre-dia first ", u, v, g)
704  endif
705 
706  if (cs%split .and. cs%legacy_split) then
707  call adjustments_dyn_legacy_split(u, v, h, dt, g, gv, cs%dyn_legacy_split_CSp)
708  endif
709 
710  call cpu_clock_begin(id_clock_diabatic)
711  call diabatic(u, v, h, cs%tv, fluxes, cs%visc, cs%ADp, cs%CDp, &
712  dtdia, g, gv, cs%diabatic_CSp)
713  fluxes%fluxes_used = .true.
714  call cpu_clock_end(id_clock_diabatic)
715 
716  if (cs%id_u_preale > 0) call post_data(cs%id_u_preale, u, cs%diag)
717  if (cs%id_v_preale > 0) call post_data(cs%id_v_preale, v, cs%diag)
718  if (cs%id_h_preale > 0) call post_data(cs%id_h_preale, h, cs%diag)
719  if (cs%id_T_preale > 0) call post_data(cs%id_T_preale, cs%tv%T, cs%diag)
720  if (cs%id_S_preale > 0) call post_data(cs%id_S_preale, cs%tv%S, cs%diag)
721  if (cs%id_e_preale > 0) then
722  call find_eta(h, cs%tv, gv%g_Earth, g, gv, eta_preale)
723  call post_data(cs%id_e_preale, eta_preale, cs%diag)
724  endif
725 
726  ! Regridding/remapping is done here, at end of thermodynamics time step
727  ! (that may comprise several dynamical time steps)
728  ! The routine 'ALE_main' can be found in 'MOM_ALE.F90'.
729  if ( cs%use_ALE_algorithm ) then
730 ! call pass_vector(u, v, G%Domain)
731  call do_group_pass(cs%pass_T_S_h, g%Domain)
732 
733  ! update squared quantities
734  if (associated(cs%S_squared)) &
735  cs%S_squared(:,:,:) = cs%tv%S(:,:,:) ** 2
736  if (associated(cs%T_squared)) &
737  cs%T_squared(:,:,:) = cs%tv%T(:,:,:) ** 2
738 
739  if (cs%debug) then
740  call mom_state_chksum("Pre-ALE 1 ", u, v, h, cs%uh, cs%vh, g, gv)
741  call hchksum(cs%tv%T,"Pre-ALE 1 T", g%HI, haloshift=1)
742  call hchksum(cs%tv%S,"Pre-ALE 1 S", g%HI, haloshift=1)
743  call check_redundant("Pre-ALE 1 ", u, v, g)
744  endif
745  call cpu_clock_begin(id_clock_ale)
746  if (use_ice_shelf) then
747 
748  call ale_main(g, gv, h, u, v, cs%tv, cs%tracer_Reg, cs%ALE_CSp, dtdia, &
749  fluxes%frac_shelf_h)
750  else
751  call ale_main(g, gv, h, u, v, cs%tv, cs%tracer_Reg, cs%ALE_CSp, dtdia)
752  endif
753 
754  call cpu_clock_end(id_clock_ale)
755  endif ! endif for the block "if ( CS%use_ALE_algorithm )"
756 
757  call cpu_clock_begin(id_clock_pass)
758  call do_group_pass(cs%pass_uv_T_S_h, g%Domain)
759  call cpu_clock_end(id_clock_pass)
760 
761  if (cs%debug .and. cs%use_ALE_algorithm) then
762  call mom_state_chksum("Post-ALE 1 ", u, v, h, cs%uh, cs%vh, g, gv)
763  call hchksum(cs%tv%T,"Post-ALE 1 T", g%HI, haloshift=1)
764  call hchksum(cs%tv%S,"Post-ALE 1 S", g%HI, haloshift=1)
765  call check_redundant("Post-ALE 1 ", u, v, g)
766  endif
767 
768  ! Whenever thickness changes let the diag manager know, target grids
769  ! for vertical remapping may need to be regenerated.
770  call diag_update_target_grids(cs%diag)
771 
772  call post_diags_ts_vardec(g, cs, dtdia)
773 
774  if (cs%debug) then
775  call uchksum(u,"Post-dia first u", g%HI, haloshift=2)
776  call vchksum(v,"Post-dia first v", g%HI, haloshift=2)
777  call hchksum(h,"Post-dia first h", g%HI, haloshift=1)
778  call uchksum(cs%uhtr,"Post-dia first uh", g%HI, haloshift=0)
779  call vchksum(cs%vhtr,"Post-dia first vh", g%HI, haloshift=0)
780  ! call MOM_state_chksum("Post-dia first ", u, v, &
781  ! h, CS%uhtr, CS%vhtr, G, GV, haloshift=1)
782  if (associated(cs%tv%T)) call hchksum(cs%tv%T, "Post-dia first T", g%HI, haloshift=1)
783  if (associated(cs%tv%S)) call hchksum(cs%tv%S, "Post-dia first S", g%HI, haloshift=1)
784  if (associated(cs%tv%frazil)) call hchksum(cs%tv%frazil, &
785  "Post-dia first frazil", g%HI, haloshift=0)
786  if (associated(cs%tv%salt_deficit)) call hchksum(cs%tv%salt_deficit, &
787  "Post-dia first salt deficit", g%HI, haloshift=0)
788  ! call MOM_thermo_chksum("Post-dia first ", CS%tv, G)
789  call check_redundant("Post-dia first ", u, v, g)
790  endif
791 
792  else ! else for block "if (.not.CS%adiabatic)"
793 
794  call cpu_clock_begin(id_clock_diabatic)
795  call adiabatic(h, cs%tv, fluxes, dtdia, g, gv, cs%diabatic_CSp)
796  fluxes%fluxes_used = .true.
797  call cpu_clock_end(id_clock_diabatic)
798 
799  if (cs%use_temperature) then
800  call cpu_clock_begin(id_clock_pass)
801  call do_group_pass(cs%pass_T_S, g%Domain)
802  call cpu_clock_end(id_clock_pass)
803  if (cs%debug) then
804  if (associated(cs%tv%T)) call hchksum(cs%tv%T, "Post-dia first T", g%HI, haloshift=1)
805  if (associated(cs%tv%S)) call hchksum(cs%tv%S, "Post-dia first S", g%HI, haloshift=1)
806  endif
807  endif
808 
809  endif ! endif for the block "if (.not.CS%adiabatic)"
810 
811  call disable_averaging(cs%diag)
812  call cpu_clock_end(id_clock_thermo)
813  if (showcalltree) call calltree_waypoint("finished diabatic_first (step_MOM)")
814 
815  endif ! end of block "(CS%diabatic_first .and. (CS%dt_trans==0.0))"
816 
817  call cpu_clock_begin(id_clock_other)
818 
819  call cpu_clock_begin(id_clock_pass)
820  if (do_pass_kd_kv_turb) call do_group_pass(cs%pass_kd_kv_turb, g%Domain)
821  call cpu_clock_end(id_clock_pass)
822 
823  call cpu_clock_end(id_clock_other)
824 
825 
826  call cpu_clock_begin(id_clock_dynamics)
827  call disable_averaging(cs%diag)
828 
829  if (cs%thickness_diffuse .and. cs%thickness_diffuse_first) then
830  if (cs%dt_trans == 0.0) then
831  if (thermo_does_span_coupling) then
832  dtth = dt_therm
833  else
834  dtth = dt*min(ntstep,n_max-n+1)
835  endif
836 
837  call enable_averaging(dtth,time_local+set_time(int(floor(dtth-dt+0.5))), cs%diag)
838  call cpu_clock_begin(id_clock_thick_diff)
839  if (associated(cs%VarMix)) &
840  call calc_slope_functions(h, cs%tv, dt, g, gv, cs%VarMix)
841  call thickness_diffuse(h, cs%uhtr, cs%vhtr, cs%tv, dtth, g, gv, &
842  cs%MEKE, cs%VarMix, cs%CDp, cs%thickness_diffuse_CSp)
843  call cpu_clock_end(id_clock_thick_diff)
844  call cpu_clock_begin(id_clock_pass)
845  call do_group_pass(cs%pass_h, g%Domain)
846  call cpu_clock_end(id_clock_pass)
847  call disable_averaging(cs%diag)
848  if (showcalltree) call calltree_waypoint("finished thickness_diffuse_first (step_MOM)")
849 
850  ! Whenever thickness changes let the diag manager know, target grids
851  ! for vertical remapping may need to be regenerated.
852  call diag_update_target_grids(cs%diag)
853 
854  endif
855  endif
856 
857  if (cs%interp_p_surf) then
858  wt_end = real(n) / real(n_max)
859  wt_beg = real(n-1) / real(n_max)
860  do j=jsd,jed ; do i=isd,ied
861  cs%p_surf_end(i,j) = wt_end * fluxes%p_surf(i,j) + &
862  (1.0-wt_end) * cs%p_surf_prev(i,j)
863  cs%p_surf_begin(i,j) = wt_beg * fluxes%p_surf(i,j) + &
864  (1.0-wt_beg) * cs%p_surf_prev(i,j)
865  enddo ; enddo
866  endif

3868 !! \section section_overview Overview of MOM
3869 !!
3870 !! This program (MOM) simulates the ocean by numerically solving
3871 !! the hydrostatic primitive equations in generalized Lagrangian
3872 !! vertical coordinates, typically tracking stretched pressure (p*)
3873 !! surfaces or following isopycnals in the ocean's interior, and
3874 !! general orthogonal horizontal coordinates. Unlike earlier versions
3875 !! of MOM, in MOM6 these equations are horizontally discretized on an
3876 !! Arakawa C-grid. (It remains to be seen whether a B-grid dynamic
3877 !! core will be revived in MOM6 at a later date; for now applications
3878 !! requiring a B-grid discretization should use MOM5.1.) MOM6 offers
3879 !! a range of options for the physical parameterizations, from those
3880 !! most appropriate to highly idealized models for geophysical fluid
3881 !! dynamics studies to a rich suite of processes appropriate for
3882 !! realistic ocean simulations. The thermodynamic options typically
3883 !! use conservative temperature and preformed salinity as conservative
3884 !! state variables and a full nonlinear equation of state, but there
3885 !! are also idealized adiabatic configurations of the model that use
3886 !! fixed density layers. Version 6.0 of MOM continues in the long
3887 !! tradition of a commitment to climate-quality ocean simulations
3888 !! embodied in previous versions of MOM, even as it draws extensively
3889 !! on the lessons learned in the development of the Generalized Ocean
3890 !! Layered Dynamics (GOLD) ocean model, which was also primarily
3891 !! developed at NOAA/GFDL. MOM has also benefited tremendously from
3892 !! the FMS infrastructure, which it utilizes and shares with other
3893 !! component models developed at NOAA/GFDL.
3894 !!
3895 !! When run is isopycnal-coordinate mode, the uppermost few layers
3896 !! are often used to describe a bulk mixed layer, including the
3897 !! effects of penetrating shortwave radiation. Either a split-
3898 !! explicit time stepping scheme or a non-split scheme may be used
3899 !! for the dynamics, while the time stepping may be split (and use
3900 !! different numbers of steps to cover the same interval) for the
3901 !! forcing, the thermodynamics, and for the dynamics. Most of the
3902 !! numerics are second order accurate in space. MOM can run with an
3903 !! absurdly thin minimum layer thickness. A variety of non-isopycnal
3904 !! vertical coordinate options are under development, but all exploit
3905 !! the advantages of a Lagrangian vertical coordinate, as discussed
3906 !! in detail by Adcroft and Hallberg (Ocean Modelling, 2006).
3907 !!
3908 !! Details of the numerics and physical parameterizations are
3909 !! provided in the appropriate source files. All of the available
3910 !! options are selected at run-time by parsing the input files,
3911 !! usually MOM_input and MOM_override, and the options choices are
3912 !! then documented for each run in MOM_param_docs.
3913 !!
3914 !! MOM6 integrates the equations forward in time in three distinct
3915 !! phases. In one phase, the dynamic equations for the velocities
3916 !! and layer thicknesses are advanced, capturing the propagation of
3917 !! external and internal inertia-gravity waves, Rossby waves, and
3918 !! other strictly adiabatic processes, including lateral stresses,
3919 !! vertical viscosity and momentum forcing, and interface height
3920 !! diffusion (commonly called Gent-McWilliams diffusion in depth-
3921 !! coordinate models). In the second phase, all tracers are advected
3922 !! and diffused along the layers. The third phase applies diabatic
3923 !! processes, vertical mixing of water properties, and perhaps
3924 !! vertical remapping to cause the layers to track the desired
3925 !! vertical coordinate.
3926 !!
3927 !! The present file (MOM.F90) orchestrates the main time stepping
3928 !! loops. One time integration option for the dynamics uses a split
3929 !! explicit time stepping scheme to rapidly step the barotropic
3930 !! pressure and velocity fields. The barotropic velocities are
3931 !! averaged over the baroclinic time step before they are used to
3932 !! advect thickness and determine the baroclinic accelerations. As
3933 !! described in Hallberg and Adcroft (2009), a barotropic correction
3934 !! is applied to the time-mean layer velocities to ensure that the
3935 !! sum of the layer transports agrees with the time-mean barotropic
3936 !! transport, thereby ensuring that the estimates of the free surface
3937 !! from the sum of the layer thicknesses agrees with the final free
3938 !! surface height as calculated by the barotropic solver. The
3939 !! barotropic and baroclinic velocities are kept consistent by
3940 !! recalculating the barotropic velocities from the baroclinic
3941 !! transports each time step. This scheme is described in Hallberg,
3942 !! 1997, J. Comp. Phys. 135, 54-65 and in Hallberg and Adcroft, 2009,
3943 !! Ocean Modelling, 29, 15-26.
3944 !!
3945 !! The other time integration options use non-split time stepping
3946 !! schemes based on the 3-step third order Runge-Kutta scheme
3947 !! described in Matsuno, 1966, J. Met. Soc. Japan, 44, 85-88, or on
3948 !! a two-step quasi-2nd order Runge-Kutta scheme. These are much
3949 !! slower than the split time-stepping scheme, but they are useful
3950 !! for providing a more robust solution for debugging cases where the
3951 !! more complicated split time-stepping scheme may be giving suspect
3952 !! solutions.
3953 !!
3954 !! There are a range of closure options available. Horizontal
3955 !! velocities are subject to a combination of horizontal biharmonic
3956 !! and Laplacian friction (based on a stress tensor formalism) and a
3957 !! vertical Fickian viscosity (perhaps using the kinematic viscosity
3958 !! of water). The horizontal viscosities may be constant, spatially
3959 !! varying or may be dynamically calculated using Smagorinsky's
3960 !! approach. A diapycnal diffusion of density and thermodynamic
3961 !! quantities is also allowed, but not required, as is horizontal
3962 !! diffusion of interface heights (akin to the Gent-McWilliams
3963 !! closure of geopotential coordinate models). The diapycnal mixing
3964 !! may use a fixed diffusivity or it may use the shear Richardson
3965 !! number dependent closure, like that described in Jackson et al.
3966 !! (JPO, 2008). When there is diapycnal diffusion, it applies to
3967 !! momentum as well. As this is in addition to the vertical viscosity,
3968 !! the vertical Prandtl always exceeds 1. A refined bulk-mixed layer
3969 !! is often used to describe the planetary boundary layer in realistic
3970 !! ocean simulations.
3971 !!
3972 !! MOM has a number of noteworthy debugging capabilities.
3973 !! Excessively large velocities are truncated and MOM will stop
3974 !! itself after a number of such instances to keep the model from
3975 !! crashing altogether. This is useful in diagnosing failures,
3976 !! or (by accepting some truncations) it may be useful for getting
3977 !! the model past the adjustment from an ill-balanced initial
3978 !! condition. In addition, all of the accelerations in the columns
3979 !! with excessively large velocities may be directed to a text file.
3980 !! Parallelization errors may be diagnosed using the DEBUG option,
3981 !! which causes extensive checksums to be written out along with
3982 !! comments indicating where in the algorithm the sums originate and
3983 !! what variable is being summed. The point where these checksums
3984 !! differ between runs is usually a good indication of where in the
3985 !! code the problem lies. All of the test cases provided with MOM
3986 !! are routinely tested to ensure that they give bitwise identical
3987 !! results regardless of the domain decomposition, or whether they
3988 !! use static or dynamic memory allocation.
3989 !!

4054 !! different compilers on GFDL's latest large computer. The versions
4055 !! of MOM_memory.h in these directories need not be used if dynamic
4056 !! memory allocation is desired, and the answers should be unchanged.

4070 !! to the model itself. While most of these diagnostics do not
4071 !! directly affect the model's solution, there are some, like the
4072 !! calculation of the deformation radius, that are used in some
4073 !! of the process parameterizations.

4077 !! including both the equation of state and when it freezes.
4078 !!

4081 !! simply wrappers for capabilities provided by FMS, although others
4082 !! provide capabilities (like the file_parser) that are unique to

4084 !! distinct from FMS, most of the required changes are in this
4085 !! directory.
