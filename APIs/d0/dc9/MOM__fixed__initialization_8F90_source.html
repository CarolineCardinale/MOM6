47 !> MOM_initialize_fixed sets up time-invariant quantities related to MOM6's
48 !! horizontal grid, bathymetry, and the Coriolis parameter.
49 subroutine mom_initialize_fixed(G, OBC, PF, write_geom, output_dir)
50  type(dyn_horgrid_type), intent(inout) :: G !< The ocean's grid structure.
51  type(ocean_obc_type), pointer :: OBC !< Open boundary structure.
52  type(param_file_type), intent(in) :: PF !< A structure indicating the open file
53  !! to parse for model parameter values.
54  logical, intent(in) :: write_geom !< If true, write grid geometry files.
55  character(len=*), intent(in) :: output_dir !< The directory into which to write files.
56 
57  ! Local
58  character(len=200) :: inputdir ! The directory where NetCDF input files are.
59  character(len=200) :: config
60  character(len=40) :: mod = "MOM_fixed_initialization" ! This module's name.
61  logical :: debug
62 ! This include declares and sets the variable "version".
63 #include "version_variable.h"
64 
65  call calltree_enter("MOM_initialize_fixed(), MOM_fixed_initialization.F90")
66  call log_version(pf, mod, version, "")
67  call get_param(pf, mod, "DEBUG", debug, default=.false.)
68 
69  call get_param(pf, mod, "INPUTDIR", inputdir, &
70  "The directory in which input files are found.", default=".")
71  inputdir = slasher(inputdir)
72 
73 ! Set up the parameters of the physical domain (i.e. the grid), G
74  call set_grid_metrics(g, pf)
75 
76 ! Set up the bottom depth, G%bathyT either analytically or from file
77 ! This also sets G%max_depth based on the input parameter MAXIMUM_DEPTH,
78 ! or, if absent, is diagnosed as G%max_depth = max( G%D(:,:) )
79  call mom_initialize_topography(g%bathyT, g%max_depth, g, pf)
80 
81  ! To initialize masks, the bathymetry in halo regions must be filled in
82  call pass_var(g%bathyT, g%Domain)
83 
84 ! Determine the position of any open boundaries
85  call open_boundary_config(g, pf, obc)
86  if (open_boundary_query(obc, apply_orig_obcs=.true.)) then
87  call get_param(pf, mod, "OBC_CONFIG", config, &
88  "A string that sets how the open boundary conditions are \n"//&
89  " configured: \n", default="none")
90  select case ( trim(config) )
91  case ("none")
92  case ("DOME") ! Avoid FATAL when using segments
93  case ("tidal_bay") ; !Using segments now
94  case ("supercritical") ; !Using segments now
95  case ("USER") ! Avoid FATAL when using segments
96  case default ; call mom_error(fatal, "MOM_initialize_fixed: "// &
97  "The open boundary positions specified by OBC_CONFIG="//&
98  trim(config)//" have not been fully implemented.")
99  end select
100  endif
101 
102  ! Make bathymetry consistent with open boundaries
103  call open_boundary_impose_normal_slope(obc, g, g%bathyT)
104 
105  ! This call sets masks that prohibit flow over any point interpreted as land
106  call initialize_masks(g, pf)
107 
108  ! Make OBC mask consistent with land mask
109  call open_boundary_impose_land_mask(obc, g, g%areaCu, g%areaCv)
110 
111  if (debug) then
112  call hchksum(g%bathyT, 'MOM_initialize_fixed: depth ', g%HI, haloshift=1)
113  call hchksum(g%mask2dT, 'MOM_initialize_fixed: mask2dT ', g%HI)
114  call uchksum(g%mask2dCu, 'MOM_initialize_fixed: mask2dCu ', g%HI)
115  call vchksum(g%mask2dCv, 'MOM_initialize_fixed: mask2dCv ', g%HI)
116  call qchksum(g%mask2dBu, 'MOM_initialize_fixed: mask2dBu ', g%HI)
117  endif
118 
119 ! Modulate geometric scales according to geography.
120  call get_param(pf, mod, "CHANNEL_CONFIG", config, &
121  "A parameter that determines which set of channels are \n"//&
122  "restricted to specific widths. Options are:\n"//&
123  " \t none - All channels have the grid width.\n"//&
124  " \t global_1deg - Sets 16 specific channels appropriate \n"//&
125  " \t\t for a 1-degree model, as used in CM2G.\n"//&
126  " \t list - Read the channel locations and widths from a \n"//&
127  " \t\t text file, like MOM_channel_list in the MOM_SIS \n"//&
128  " \t\t test case.\n"//&
129  " \t file - Read open face widths everywhere from a \n"//&
130  " \t\t NetCDF file on the model grid.", &
131  default="none")
132  select case ( trim(config) )
133  case ("none")
134  case ("list") ; call reset_face_lengths_list(g, pf)
135  case ("file") ; call reset_face_lengths_file(g, pf)
136  case ("global_1deg") ; call reset_face_lengths_named(g, pf, trim(config))
137  case default ; call mom_error(fatal, "MOM_initialize_fixed: "// &
138  "Unrecognized channel configuration "//trim(config))
139  end select
140 
141 ! This call sets the topography at velocity points.
142  if (g%bathymetry_at_vel) then
143  call get_param(pf, mod, "VELOCITY_DEPTH_CONFIG", config, &
144  "A string that determines how the topography is set at \n"//&
145  "velocity points. This may be 'min' or 'max'.", &
146  default="max")
147  select case ( trim(config) )
148  case ("max") ; call set_velocity_depth_max(g)
149  case ("min") ; call set_velocity_depth_min(g)
150  case default ; call mom_error(fatal, "MOM_initialize_fixed: "// &
151  "Unrecognized velocity depth configuration "//trim(config))
152  end select
153  endif
154 
155 ! Calculate the value of the Coriolis parameter at the latitude !
156 ! of the q grid points, in s-1.
157  call mom_initialize_rotation(g%CoriolisBu, g, pf)
158 ! Calculate the components of grad f (beta)
159  call mom_calculate_grad_coriolis(g%dF_dx, g%dF_dy, g)
160  if (debug) then
161  call qchksum(g%CoriolisBu, "MOM_initialize_fixed: f ", g%HI)
162  call hchksum(g%dF_dx, "MOM_initialize_fixed: dF_dx ", g%HI)
163  call hchksum(g%dF_dy, "MOM_initialize_fixed: dF_dy ", g%HI)
164  endif
165 
166 ! Compute global integrals of grid values for later use in scalar diagnostics !
167  call compute_global_grid_integrals(g)
168 
169 ! Write out all of the grid data used by this run.
170  if (write_geom) call write_ocean_geometry_file(g, pf, output_dir)
171 
172  call calltree_leave('MOM_initialize_fixed()')
173 
174 end subroutine mom_initialize_fixed

176 !> MOM_initialize_topography makes the appropriate call to set up the bathymetry.
177 subroutine mom_initialize_topography(D, max_depth, G, PF)
178  type(dyn_horgrid_type), intent(in) :: G !< The dynamic horizontal grid type
179  real, dimension(G%isd:G%ied,G%jsd:G%jed), &
180  intent(out) :: D !< Ocean bottom depth in m
181  type(param_file_type), intent(in) :: PF !< Parameter file structure
182  real, intent(out) :: max_depth !< Maximum depth of model in m
183 
184 ! This subroutine makes the appropriate call to set up the bottom depth.
185 ! This is a separate subroutine so that it can be made public and shared with
186 ! the ice-sheet code or other components.
187 ! Set up the bottom depth, G%bathyT either analytically or from file
188  character(len=40) :: mod = "MOM_initialize_topography" ! This subroutine's name.
189  character(len=200) :: config
190 
191  call get_param(pf, mod, "TOPO_CONFIG", config, &
192  "This specifies how bathymetry is specified: \n"//&
193  " \t file - read bathymetric information from the file \n"//&
194  " \t\t specified by (TOPO_FILE).\n"//&
195  " \t flat - flat bottom set to MAXIMUM_DEPTH. \n"//&
196  " \t bowl - an analytically specified bowl-shaped basin \n"//&
197  " \t\t ranging between MAXIMUM_DEPTH and MINIMUM_DEPTH. \n"//&
198  " \t spoon - a similar shape to 'bowl', but with an vertical \n"//&
199  " \t\t wall at the southern face. \n"//&
200  " \t halfpipe - a zonally uniform channel with a half-sine \n"//&
201  " \t\t profile in the meridional direction. \n"//&
202  " \t benchmark - use the benchmark test case topography. \n"//&
203  " \t DOME - use a slope and channel configuration for the \n"//&
204  " \t\t DOME sill-overflow test case. \n"//&
205  " \t ISOMIP - use a slope and channel configuration for the \n"//&
206  " \t\t ISOMIP test case. \n"//&
207  " \t DOME2D - use a shelf and slope configuration for the \n"//&
208  " \t\t DOME2D gravity current/overflow test case. \n"//&
209  " \t seamount - Gaussian bump for spontaneous motion test case.\n"//&
210  " \t supercritical - flat but with 8.95 degree land mask.\n"//&
211  " \t Phillips - ACC-like idealized topography used in the Phillips config.\n"//&
212  " \t USER - call a user modified routine.", &
213  fail_if_missing=.true.)
214  max_depth = -1.e9; call read_param(pf, "MAXIMUM_DEPTH", max_depth)
215  select case ( trim(config) )
216  case ("file"); call initialize_topography_from_file(d, g, pf)
217  case ("flat"); call initialize_topography_named(d, g, pf, config, max_depth)
218  case ("spoon"); call initialize_topography_named(d, g, pf, config, max_depth)
219  case ("bowl"); call initialize_topography_named(d, g, pf, config, max_depth)
220  case ("halfpipe"); call initialize_topography_named(d, g, pf, config, max_depth)
221  case ("DOME"); call dome_initialize_topography(d, g, pf, max_depth)
222  case ("ISOMIP"); call isomip_initialize_topography(d, g, pf, max_depth)
223  case ("benchmark"); call benchmark_initialize_topography(d, g, pf, max_depth)
224  case ("DOME2D"); call dome2d_initialize_topography(d, g, pf, max_depth)
225  case ("sloshing"); call sloshing_initialize_topography(d, g, pf, max_depth)
226  case ("seamount"); call seamount_initialize_topography(d, g, pf, max_depth)
227  case ("Phillips"); call phillips_initialize_topography(d, g, pf, max_depth)
228  case ("supercritical"); call supercritical_initialize_topography(d, g, pf, max_depth)
229  case ("USER"); call user_initialize_topography(d, g, pf, max_depth)
230  case default ; call mom_error(fatal,"MOM_initialize_topography: "// &
231  "Unrecognized topography setup '"//trim(config)//"'")
232  end select
233  if (max_depth>0.) then
234  call log_param(pf, mod, "MAXIMUM_DEPTH", max_depth, &
235  "The maximum depth of the ocean.", units="m")
236  else
237  max_depth = diagnosemaximumdepth(d,g)
238  call log_param(pf, mod, "!MAXIMUM_DEPTH", max_depth, &
239  "The (diagnosed) maximum depth of the ocean.", units="m")
240  endif
241  if (trim(config) .ne. "DOME") then
242  call limit_topography(d, g, pf, max_depth)
243  endif
244 
245 end subroutine mom_initialize_topography
