46 !> Subroutine for calculating (internal) diffusivity
47 subroutine calculate_cvmix_shear(u_H, v_H, h, tv, KH, &
48  KM, G, GV, CS )
49  type(ocean_grid_type), intent(in) :: G !< Grid structure.
50  type(verticalgrid_type), intent(in) :: GV !< Vertical grid structure.
51  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: u_H !< Initial zonal velocity on T points, in m s-1.
52  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: v_H !< Initial meridional velocity on T points, in m s-1.
53  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< Layer thickness, in m or kg m-2.
54  type(thermo_var_ptrs), intent(in) :: tv !< Thermodynamics structure.
55  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(out) :: KH !< The vertical viscosity at each interface
56  !! (not layer!) in m2 s-1.
57  real, dimension(SZI_(G),SZJ_(G),SZK_(G)+1), intent(out) :: KM !< The vertical viscosity at each interface
58  !! (not layer!) in m2 s-1.
59  type(cvmix_shear_cs), pointer :: CS !< The control structure returned by a previous call to
60  !! CVMix_shear_init.
61  ! Local variables
62  integer :: i, j, k, kk, km1
63  real :: gorho
64  real :: pref, DU, DV, DRHO, DZ, N2, S2
65  real, dimension(2*(G%ke)) :: pres_1d, temp_1d, salt_1d, rho_1d
66  real, dimension(G%ke+1) :: Ri_Grad !< Gradient Richardson number
67 
68  ! some constants
69  gorho = gv%g_Earth / gv%Rho0
70 
71  do j = g%jsc, g%jec
72  do i = g%isc, g%iec
73 
74  ! skip calling for land points
75  if (g%mask2dT(i,j)==0.) cycle
76 
77  ! Richardson number computed for each cell in a column.
78  pref = 0.
79  ri_grad(:)=1.e8 !Initialize w/ large Richardson value
80  do k=1,g%ke
81  ! pressure, temp, and saln for EOS
82  ! kk+1 = k fields
83  ! kk+2 = km1 fields
84  km1 = max(1, k-1)
85  kk = 2*(k-1)
86  pres_1d(kk+1) = pref
87  pres_1d(kk+2) = pref
88  temp_1d(kk+1) = tv%T(i,j,k)
89  temp_1d(kk+2) = tv%T(i,j,km1)
90  salt_1d(kk+1) = tv%S(i,j,k)
91  salt_1d(kk+2) = tv%S(i,j,km1)
92 
93  ! pRef is pressure at interface between k and km1.
94  ! iterate pRef for next pass through k-loop.
95  pref = pref + gv%H_to_Pa * h(i,j,k)
96 
97  enddo ! k-loop finishes
98 
99  ! compute in-situ density
100  call calculate_density(temp_1d, salt_1d, pres_1d, rho_1d, 1, 2*g%ke, tv%EQN_OF_STATE)
101 
102  ! N2 (can be negative) on interface
103  do k = 1, g%ke
104  km1 = max(1, k-1)
105  kk = 2*(k-1)
106  du = (u_h(i,j,k))-(u_h(i,j,km1))
107  dv = (v_h(i,j,k))-(v_h(i,j,km1))
108  drho = (gorho * (rho_1d(kk+1) - rho_1d(kk+2)) )
109  dz = ((0.5*(h(i,j,km1) + h(i,j,k))+gv%H_subroundoff)*gv%H_to_m)
110  n2 = drho/dz
111  s2 = (du*du+dv*dv)/(dz*dz)
112  ri_grad(k) = max(0.,n2)/max(s2,1.e-16)
113  enddo
114 
115  ! Call to CVMix wrapper for computing interior mixing coefficients.
116  call cvmix_coeffs_shear(mdiff_out=km(i,j,:), &
117  tdiff_out=kh(i,j,:), &
118  rich=ri_grad, &
119  nlev=g%ke, &
120  max_nlev=g%ke)
121  enddo
122  enddo
123 
124 end subroutine calculate_cvmix_shear

127 !> Initialized the cvmix internal shear mixing routine.
128 !! \note *This is where we test to make sure multiple internal shear
129 !! mixing routines (including JHL) are not enabled at the same time.
130 !! (returns) cvmix_shear_init - True if module is to be used, False otherwise
131 logical function cvmix_shear_init(Time, G, GV, param_file, diag, CS)
132  type(time_type), intent(in) :: Time !< The current time.
133  type(ocean_grid_type), intent(in) :: G !< Grid structure.
134  type(verticalgrid_type), intent(in) :: GV !< Vertical grid structure.
135  type(param_file_type), intent(in) :: param_file !< Run-time parameter file handle
136  type(diag_ctrl), target, intent(inout) :: diag !< Diagnostics control structure.
137  type(cvmix_shear_cs), pointer :: CS !< This module's control structure.
138  ! Local variables
139  integer :: NumberTrue=0
140  logical :: use_JHL
141 ! This include declares and sets the variable "version".
142 #include "version_variable.h"
143 
144  if (associated(cs)) then
145  call mom_error(warning, "cvmix_shear_init called with an associated "// &
146  "control structure.")
147  return
148  endif
149  allocate(cs)
150 
151 ! Set default, read and log parameters
152  call log_version(param_file, mod, version, &
153  "Parameterization of shear-driven turbulence via CVMix (various options)")
154  call get_param(param_file, mod, "USE_LMD94", cs%use_LMD94, &
155  "If true, use the Large-McWilliams-Doney (JGR 1994) \n"//&
156  "shear mixing parameterization.", default=.false.)
157  if (cs%use_LMD94) then
158  numbertrue=numbertrue + 1
159  cs%Mix_Scheme='KPP'
160  endif
161  call get_param(param_file, mod, "USE_PP81", cs%use_PP81, &
162  "If true, use the Pacanowski and Philander (JPO 1981) \n"//&
163  "shear mixing parameterization.", default=.false.)
164  if (cs%use_PP81) then
165  numbertrue = numbertrue + 1
166  cs%Mix_Scheme='PP'
167  endif
168  use_jhl=kappa_shear_is_used(param_file)
169  if (use_jhl) numbertrue = numbertrue + 1
170  ! After testing for interior schemes, make sure only 0 or 1 are enabled.
171  ! Otherwise, warn user and kill job.
172  if ((numbertrue).gt.1) then
173  call mom_error(fatal, 'MOM_cvmix_shear_init: '// &
174  'Multiple shear driven internal mixing schemes selected,'//&
175  ' please disable all but one scheme to proceed.')
176  endif
177  cvmix_shear_init=(cs%use_PP81.or.cs%use_LMD94)
178 
179 ! Forego remainder of initialization if not using this scheme
180  if (.not. cvmix_shear_init) return
181  call get_param(param_file, mod, "NU_ZERO", cs%Nu_Zero, &
182  "Leading coefficient in KPP shear mixing.", &
183  units="nondim", default=5.e-3)
184  call get_param(param_file, mod, "RI_ZERO", cs%Ri_Zero, &
185  "Critical Richardson for KPP shear mixing,"// &
186  " NOTE this the internal mixing and this is"// &
187  " not for setting the boundary layer depth." &
188  ,units="nondim", default=0.7)
189  call get_param(param_file, mod, "KPP_EXP", cs%KPP_exp, &
190  "Exponent of unitless factor of diffusivities,"// &
191  " for KPP internal shear mixing scheme." &
192  ,units="nondim", default=3.0)
193  call cvmix_init_shear(mix_scheme=cs%mix_scheme, &
194  kpp_nu_zero=cs%Nu_Zero, &
195  kpp_ri_zero=cs%Ri_zero, &
196  kpp_exp=cs%KPP_exp)
197  ! Allocation and initialization
198  allocate( cs%N2( szi_(g), szj_(g), szk_(g)+1 ) );cs%N2(:,:,:) = 0.
199  allocate( cs%S2( szi_(g), szj_(g), szk_(g)+1 ) );cs%S2(:,:,:) = 0.
200 
201 end function cvmix_shear_init
