134 !> Enables OBC module and reads configuration parameters
135 subroutine open_boundary_config(G, param_file, OBC)
136  type(dyn_horgrid_type), intent(in) :: G !< Ocean grid structure
137  type(param_file_type), intent(in) :: param_file !< Parameter file handle
138  type(ocean_obc_type), pointer :: OBC !< Open boundary control structure
139  ! Local variables
140  integer :: l ! For looping over segments
141  character(len=15) :: segment_param_str ! The run-time parameter name for each segment
142  character(len=100) :: segment_str ! The contents (rhs) for parameter "segment_param_str"
143 
144  allocate(obc)
145 
146  call log_version(param_file, mod, version, "Controls where open boundaries are located, what "//&
147  "kind of boundary condition to impose, and what data to apply, if any.")
148  call get_param(param_file, mod, "OBC_NUMBER_OF_SEGMENTS", obc%number_of_segments, &
149  "The number of open boundary segments.", &
150  default=0)
151  if (obc%number_of_segments > 0) then
152  ! Allocate everything
153  allocate(obc%OBC_segment_list(0:obc%number_of_segments))
154  do l=0,obc%number_of_segments
155  obc%OBC_segment_list(l)%Flather = .false.
156  obc%OBC_segment_list(l)%radiation = .false.
157  obc%OBC_segment_list(l)%radiation2D = .false.
158  obc%OBC_segment_list(l)%nudged = .false.
159  obc%OBC_segment_list(l)%specified = .false.
160  obc%OBC_segment_list(l)%gradient = .false.
161  obc%OBC_segment_list(l)%direction = obc_none
162  obc%OBC_segment_list(l)%Tnudge_in = 0.0
163  obc%OBC_segment_list(l)%Tnudge_out = 0.0
164  enddo
165  allocate(obc%OBC_mask_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; obc%OBC_mask_u(:,:) = .false.
166  allocate(obc%OBC_direction_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; obc%OBC_direction_u(:,:) = obc_none
167  allocate(obc%OBC_segment_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; obc%OBC_segment_u(:,:) = obc_none
168  allocate(obc%OBC_mask_v(g%isd:g%ied,g%JsdB:g%JedB)) ; obc%OBC_mask_v(:,:) = .false.
169  allocate(obc%OBC_direction_v(g%isd:g%ied,g%JsdB:g%JedB)) ; obc%OBC_direction_v(:,:) = obc_none
170  allocate(obc%OBC_segment_v(g%isd:g%ied,g%JsdB:g%JedB)) ; obc%OBC_segment_v(:,:) = obc_none
171 
172  do l = 1, obc%number_of_segments
173  write(segment_param_str(1:15),"('OBC_SEGMENT_',i3.3)") l
174  call get_param(param_file, mod, segment_param_str, segment_str, &
175  "Documentation needs to be dynamic?????", &
176  fail_if_missing=.true.)
177  segment_str = remove_spaces(segment_str)
178  if (segment_str(1:2) == 'I=') then
179  call setup_u_point_obc(obc, g, segment_str, l)
180  elseif (segment_str(1:2) == 'J=') then
181  call setup_v_point_obc(obc, g, segment_str, l)
182  else
183  call mom_error(fatal, "MOM_open_boundary.F90, open_boundary_config: "//&
184  "Unable to interpret "//segment_param_str//" = "//trim(segment_str))
185  endif
186  enddo
187  endif
188 
189  ! Safety check
190  if ((obc%Flather_u_BCs_exist_globally .or. obc%Flather_v_BCs_exist_globally) .and. &
191  .not.g%symmetric ) call mom_error(fatal, &
192  "MOM_open_boundary, open_boundary_config: "//&
193  "Symmetric memory must be used when using Flather OBCs.")
194 
195  if (.not.(obc%specified_u_BCs_exist_globally .or. obc%specified_v_BCs_exist_globally .or. &
196  obc%Flather_u_BCs_exist_globally .or. obc%Flather_v_BCs_exist_globally)) then
197  ! No open boundaries have been requested
198  call open_boundary_dealloc(obc)
199  endif
200 
201 end subroutine open_boundary_config

203 !> Parse an OBC_SEGMENT_%%% string starting with "I=" and configure placement and type of OBC accordingly
204 subroutine setup_u_point_obc(OBC, G, segment_str, l_seg)
205  type(ocean_obc_type), pointer :: OBC !< Open boundary control structure
206  type(dyn_horgrid_type), intent(in) :: G !< Ocean grid structure
207  character(len=*), intent(in) :: segment_str !< A string in form of "I=%,J=%:%,string"
208  integer, intent(in) :: l_seg !< which segment is this?
209  ! Local variables
210  integer :: I_obc, Js_obc, Je_obc ! Position of segment in global index space
211  integer :: j, this_kind, a_loop
212  character(len=32) :: action_str(5)
213 
214  ! This returns the global indices for the segment
215  call parse_segment_str(g%ieg, g%jeg, segment_str, i_obc, js_obc, je_obc, action_str )
216  i_obc = i_obc - g%idg_offset ! Convert to local tile indices on this tile
217  js_obc = js_obc - g%jdg_offset ! Convert to local tile indices on this tile
218  je_obc = je_obc - g%jdg_offset ! Convert to local tile indices on this tile
219  this_kind = obc_none
220 
221  ! Hack to extend segment by one point
222  if (js_obc<je_obc) then
223  js_obc = js_obc - 1 ; je_obc = je_obc + 1
224  else
225  js_obc = js_obc + 1 ; je_obc = je_obc - 1
226  endif
227 
228  if (je_obc>js_obc) obc%OBC_segment_list(l_seg)%direction = obc_direction_e
229  if (je_obc<js_obc) obc%OBC_segment_list(l_seg)%direction = obc_direction_w
230  do a_loop = 1,5
231  if (len_trim(action_str(a_loop)) == 0) then
232  cycle
233  elseif (trim(action_str(a_loop)) == 'FLATHER') then
234  this_kind = obc_flather
235  obc%OBC_segment_list(l_seg)%Flather = .true.
236  ! This is a total hack for the tangential flow! - KSH
237  obc%OBC_segment_list(l_seg)%gradient = .true.
238  obc%Flather_u_BCs_exist_globally = .true.
239  elseif (trim(action_str(a_loop)) == 'ORLANSKI') then
240  obc%OBC_segment_list(l_seg)%radiation = .true.
241  ! This is a total hack for the tangential flow! - KSH
242  obc%OBC_segment_list(l_seg)%gradient = .true.
243  obc%Flather_u_BCs_exist_globally = .true.
244  elseif (trim(action_str(a_loop)) == 'OBLIQUE') then
245  obc%OBC_segment_list(l_seg)%radiation = .true.
246  obc%OBC_segment_list(l_seg)%radiation2D = .true.
247  obc%Flather_u_BCs_exist_globally = .true.
248  elseif (trim(action_str(a_loop)) == 'NUDGED') then
249  obc%OBC_segment_list(l_seg)%nudged = .true.
250  obc%nudged_uv_BCs_exist_globally = .true.
251  elseif (trim(action_str(a_loop)) == 'SIMPLE') then
252  obc%OBC_segment_list(l_seg)%specified = .true.
253  obc%specified_u_BCs_exist_globally = .true. ! This avoids deallocation
254  ! Hack to undo the hack above for SIMPLE BCs
255  if (js_obc<je_obc) then
256  js_obc = js_obc + 1 ; je_obc = je_obc - 1
257  else
258  js_obc = js_obc - 1 ; je_obc = je_obc + 1
259  endif
260  else
261  call mom_error(fatal, "MOM_open_boundary.F90, setup_u_point_obc: "//&
262  "String '"//trim(action_str(a_loop))//"' not understood.")
263  endif
264 
265  if (i_obc<g%HI%IsdB .or. i_obc>g%HI%IedB) return ! Boundary is not on tile
266  if (max(js_obc,je_obc)<g%HI%JsdB .or. min(js_obc,je_obc)>g%HI%JedB) return ! Segment is not on tile
267 
268  do j=g%HI%jsd, g%HI%jed
269  if (j>min(js_obc,je_obc) .and. j<=max(js_obc,je_obc)) then
270  obc%OBC_mask_u(i_obc,j) = .true.
271  obc%OBC_segment_u(i_obc,j) = l_seg
272  if (je_obc>js_obc) then ! East is outward
273  if (this_kind == obc_flather) then
274  obc%OBC_direction_u(i_obc,j) = obc_direction_e ! We only use direction for Flather (maybe)
275  ! Set v points outside segment
276  obc%OBC_mask_v(i_obc+1,j) = .true.
277  if (obc%OBC_segment_v(i_obc+1,j) == obc_none) then
278  obc%OBC_direction_v(i_obc+1,j) = obc_direction_e
279  obc%OBC_segment_v(i_obc+1,j) = l_seg
280  endif
281  obc%OBC_mask_v(i_obc+1,j-1) = .true.
282  if (obc%OBC_segment_v(i_obc+1,j-1) == obc_none) then
283  obc%OBC_direction_v(i_obc+1,j-1) = obc_direction_e
284  obc%OBC_segment_v(i_obc+1,j-1) = l_seg
285  endif
286  endif
287  else ! West is outward
288  if (this_kind == obc_flather) then
289  obc%OBC_direction_u(i_obc,j) = obc_direction_w ! We only use direction for Flather
290  ! Set v points outside segment
291  obc%OBC_mask_v(i_obc,j) = .true.
292  if (obc%OBC_segment_v(i_obc,j) == obc_none) then
293  obc%OBC_direction_v(i_obc,j) = obc_direction_w
294  obc%OBC_segment_v(i_obc,j) = l_seg
295  endif
296  obc%OBC_mask_v(i_obc,j-1) = .true.
297  if (obc%OBC_segment_v(i_obc,j-1) == obc_none) then
298  obc%OBC_direction_v(i_obc,j-1) = obc_direction_w
299  obc%OBC_segment_v(i_obc,j-1) = l_seg
300  endif
301  endif
302  endif
303  endif
304  enddo
305  enddo ! a_loop
306 
307 end subroutine setup_u_point_obc
